class Solution 
{
public:
    using pii=pair<int, int>;
    #定义别名pii，指向pair<int, int>类型
    vector<int> getFinalState(vector<int>& nums,int k,int multiplier)
    {   
        vector<int> res(num.size());
        #vector动态分配内存
        #设置初始值，有利与减少分配操作，但是如果初始值设置过大，会浪费内存空间。
        #分配操作过多可能影响内存的连续性

        priority_queue<pii,vector<pii>,greater<>> pq;
        for(int i=0;i<nums.size();i++)
        {
            pq.emplace(nums[i],i);

        }

        while(k--)
        {
            auto t=pq.top();
            pq.pop();
            t.first *= multiplier;
            pq.push(t);

        }
        // 将队列中的元素存储到结果数组中
        while (!pq.empty()) {
            auto [val, pos] = pq.top();
            pq.pop();
            res[pos] = val;
        }
        return res;
    }
};
